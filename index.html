from __future__ import annotations

import json
import os
import time
from dataclasses import dataclass, field
from typing import Any, Callable, Dict, Iterable, List, Optional, Tuple

import anthropic

# =====================
# Config & Environment
# =====================
API_KEY = os.environ.get("ANTHROPIC_API_KEY")
MODEL = os.environ.get("ANTHROPIC_MODEL_ID", "claude-3-5-sonnet-latest")

if not API_KEY:
    raise RuntimeError("ANTHROPIC_API_KEY is not set in the environment")

client = anthropic.Anthropic(api_key=API_KEY)


# =====================
# Types (lightweight)
# =====================
TextBlock = Dict[str, Any]  # expects {"type": "text", "text": "..."}
ToolUseBlock = Dict[str, Any]  # expects {"type": "tool_use", "id": str, "name": str, "input": any}
ToolResultBlock = Dict[str, Any]  # {"type":"tool_result", "tool_use_id": str, "content": str|list|dict}
Message = Dict[str, Any]


# =====================
# Helpers
# =====================

def extract_text(blocks: List[Dict[str, Any]]) -> str:
    """Concatenate all text blocks from a message.content list."""
    parts: List[str] = []
    for b in blocks or []:
        if b.get("type") == "text":
            parts.append(b.get("text", ""))
    return "".join(parts)


def user_msg(content: str) -> Message:
    return {"role": "user", "content": content}


def tool_result_msg(name: str, tool_use_id: str, content: Any, is_error: bool = False) -> Message:
    return {
        "role": "tool",
        "name": name,
        "tool_use_id": tool_use_id,
        "content": json.dumps(content) if not isinstance(content, str) else content,
        "is_error": is_error or None,
    }


def with_retries(fn: Callable[[], Any], attempts: int = 3, base_ms: int = 400) -> Any:
    last_err: Optional[BaseException] = None
    for i in range(attempts):
        try:
            return fn()
        except BaseException as e:  # capture SDK/network errors
            last_err = e
            if i == attempts - 1:
                break
            time.sleep(base_ms / 1000 * (2 ** i))
    assert last_err is not None
    raise last_err


# =====================
# Core request wrappers
# =====================

def send_message(messages: List[Message], max_tokens: int = 1024) -> anthropic.types.Message:
    """Single call convenience wrapper."""
    return with_retries(
        lambda: client.messages.create(
            model=MODEL,
            max_tokens=max_tokens,
            messages=messages,
        )
    )


def ask(prompt: str, *, max_tokens: int = 1024) -> str:
    resp = send_message([user_msg(prompt)], max_tokens=max_tokens)
    return extract_text(resp.content)  # type: ignore[arg-type]


# =====================
# Streaming (SSE)
# =====================

def stream_text(prompt: str, *, max_tokens: int = 1024) -> str:
    """Stream tokens to stdout and return the final text (if available)."""
    # The Python SDK provides a streaming context manager.
    with client.messages.stream(
        model=MODEL,
        max_tokens=max_tokens,
        messages=[user_msg(prompt)],
    ) as stream:
        for event in stream:
            if event.type == "content_block_delta" and getattr(event, "delta", None):
                delta = event.delta
                if getattr(delta, "type", None) == "text_delta":
                    # Write streamed text as it arrives
                    print(delta.text, end="", flush=True)
        # Get the final assembled message (if provided by SDK version)
        try:
            final = stream.get_final_message()
            return extract_text(final.content)  # type: ignore[arg-type]
        except Exception:
            return ""


# =====================
# Conversation Manager
# =====================
@dataclass
class ConversationManager:
    history: List[Message] = field(default_factory=list)

    def add_system(self, content: str) -> None:
        self.history.append({"role": "system", "content": content})

    def chat(self, user_message: str, *, max_tokens: int = 2048) -> str:
        self.history.append(user_msg(user_message))
        res = send_message(self.history, max_tokens=max_tokens)
        text = extract_text(res.content)  # type: ignore[arg-type]
        self.history.append({"role": "assistant", "content": text})
        return text


# =====================
# Generic Tool System (multi-call loop)
# =====================
ToolHandler = Callable[[Any], Any]

@dataclass
class Tool:
    name: str
    description: str
    input_schema: Dict[str, Any]
    handler: ToolHandler


def to_tool_spec(tools: Iterable[Tool]) -> List[Dict[str, Any]]:
    return [
        {
            "name": t.name,
            "description": t.description,
            "input_schema": t.input_schema,
        }
        for t in tools
    ]


def run_tools_loop(prompt: str, tools: Iterable[Tool], *, max_tokens: int = 2048) -> str:
    """Allow the model to call tools; execute them; return final assistant text.

    - Supports multiple sequential tool calls in a single round.
    - You can adapt this to maintain longer histories if desired.
    """
    tools = list(tools)
    tool_map: Dict[str, Tool] = {t.name: t for t in tools}

    # 1) Initial request advertising available tools
    response = with_retries(
        lambda: client.messages.create(
            model=MODEL,
            max_tokens=max_tokens,
            tools=to_tool_spec(tools),
            messages=[user_msg(prompt)],
        )
    )

    # 2) While the assistant requests tool_use, resolve them in order
    msgs: List[Message] = [user_msg(prompt)]
    while True:
        tool_calls: List[ToolUseBlock] = [
            b for b in response.content  # type: ignore[attr-defined]
            if isinstance(b, dict) and b.get("type") == "tool_use"
        ]

        if not tool_calls:
            break

        for call in tool_calls:
            name = str(call.get("name"))
            tool = tool_map.get(name)
            if not tool:
                # unknown tool; return an error result
                msgs.append(tool_result_msg(name, str(call.get("id")), {"error": "Unknown tool"}, is_error=True))
                continue
            try:
                result = tool.handler(call.get("input"))
                msgs.append(tool_result_msg(name, str(call.get("id")), result))
            except Exception as e:
                msgs.append(tool_result_msg(name, str(call.get("id")), {"error": str(e)}, is_error=True))

        # 3) Send accumulated tool results back to the model
        response = with_retries(
            lambda: client.messages.create(
                model=MODEL,
                max_tokens=max_tokens,
                tools=to_tool_spec(tools),
                messages=msgs,
            )
        )

        # If the assistant returns more tool_use blocks, continue the loop.

    return extract_text(response.content)  # type: ignore[arg-type]


# =====================
# Example generic tools (safe & simple)
# =====================

def tool_get_time(_: Any) -> Dict[str, str]:
    import datetime as _dt
    now = _dt.datetime.now(_dt.timezone.utc).isoformat()
    return {"now_utc": now}


def tool_echo(input: Any) -> Any:
    return {"echo": input}


# Predefined tool instances (optional)
GET_TIME_TOOL = Tool(
    name="get_time",
    description="Returns the current UTC timestamp in ISO 8601 format.",
    input_schema={"type": "object", "properties": {}, "additionalProperties": False},
    handler=tool_get_time,
)

ECHO_TOOL = Tool(
    name="echo",
    description="Echos back any JSON given as input.",
    input_schema={"type": "object", "additionalProperties": True},
    handler=tool_echo,
)


# =====================
# Batch utility (generic templating)
# =====================

def batch_process(items: Iterable[Any], template: str, *, max_tokens: int = 512) -> List[str]:
    """Apply a prompt template to each item.
    The template should contain "{item}" placeholder.
    """
    outputs: List[str] = []
    for item in items:
        prompt = template.format(item=item)
        outputs.append(ask(prompt, max_tokens=max_tokens))
    return outputs
